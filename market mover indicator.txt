//@version=6
indicator("Institutional Market Mover Dashboard v3", overlay=true)

// ==========================================
// USER INPUTS: Fundamental & Sentiment Filters (Factors 1-6, 9-12)
// ==========================================
var string G_MACRO  = "1. Macro & Sentiment Bias (Manual Overrides)"
bool input_hawkish  = input.bool(true, "Central Bank Bias (Checked = Hawkish, Unchecked = Dovish)", group=G_MACRO)
string input_sent   = input.string("Risk-On", "Overall Market Sentiment", options=["Risk-On", "Risk-Off"], group=G_MACRO)
bool input_cot      = input.bool(true, "Institutional COT Alignment (Checked = Long, Unchecked = Short)", group=G_MACRO)

var string G_PARAMS = "2. Technical & Liquidity Parameters"
int trend_ema_fast  = input.int(50, "Fast Trend EMA", group=G_PARAMS)
int trend_ema_slow  = input.int(200, "Slow Trend EMA", group=G_PARAMS)
int vol_lookback    = input.int(20, "Volume/Volatility Lookback", group=G_PARAMS)

// ==========================================
// CORE FACTOR CALCULATIONS
// ==========================================

// Factors 14 & 15: Daily & Weekly Trend Direction (Approximated via HTF EMAs)
ema_fast = ta.ema(close, trend_ema_fast)
ema_slow = ta.ema(close, trend_ema_slow)
bool daily_bullish = close > ema_fast and ema_fast > ema_slow

// Factor 16 & 17: Market Structure & Liquidity Pools
var float pdh = na
var float pdl = na
if ta.change(time("D")) != 0
    pdh := request.security(syminfo.tickerid, "D", high[1], lookahead=barmerge.lookahead_on)
    pdl := request.security(syminfo.tickerid, "D", low[1], lookahead=barmerge.lookahead_on)

// Factors 18 & 19: Liquidity Sweeps & Order Blocks
bool sweep_high = high > pdh and close < pdh
bool sweep_low  = low < pdl and close > pdl

// Factors 23 & 24: Volume Spikes & Volatility Expansion
avg_vol  = ta.sma(volume, vol_lookback)
avg_atr  = ta.atr(vol_lookback)
bool vol_spike   = volume > (avg_vol * 1.5)

// Factor 25: Correlated Asset Movement (e.g., US Dollar Index DXY correlation)
dxy_close = request.security("TVC:DXY", timeframe.period, close)
dxy_change = ta.change(dxy_close)
bool correlation_aligned = (dxy_change < 0 and close > close[1]) or (dxy_change > 0 and close < close[1])

// ==========================================
// THE COLLECTIVE CONSOLIDATION ALGORITHM
// ==========================================
int bullish_score = 0
int bearish_score = 0

bullish_score += input_hawkish ? 2 : 0
bearish_score += input_hawkish ? 0 : 2

bullish_score += (input_sent == "Risk-On") ? 1 : 0
bearish_score += (input_sent == "Risk-On") ? 0 : 1

bullish_score += input_cot ? 2 : 0
bearish_score += input_cot ? 0 : 2

bullish_score += daily_bullish ? 2 : 0
bearish_score += daily_bullish ? 0 : 2

if sweep_low and vol_spike
    bullish_score += 3  
if sweep_high and vol_spike
    bearish_score += 3 

bullish_score += correlation_aligned ? 1 : 0
bearish_score += correlation_aligned ? 0 : 1

// Determine Final Consensus and Pip Projection Range
string bias_report = ""
string direction_text = ""
float projected_pips = 0.0

float pip_value = syminfo.mintick * 10
float current_atr_pips = (ta.atr(14) / pip_value)
float expected_move = math.round(current_atr_pips * 3)

if bullish_score > bearish_score + 2
    bias_report := "BULLISH CONVERGENCE:\nMacro hawkishness\ncombined with institutional\nliquidity sweeps and a positive\ncorrelation matrix indicates\nmarket movers are aggressively\naccumulating positions,\ntargeting upper resting\nliquidity pools safely."
    projected_pips := expected_move
    direction_text := "TO THE UPSIDE\n(BULLISH)"
else if bearish_score > bullish_score + 2
    bias_report := "BEARISH CONVERGENCE:\nMacro dovishness and\nstructural breakdowns\nconfirmed by volume-backed\nliquidity sweeps indicate\nmarket movers are distributing\npositions, targeting lower\nretail stop losses."
    projected_pips := expected_move
    direction_text := "TO THE DOWNSIDE\n(BEARISH)"
else
    bias_report := "NEUTRAL CHOP:\nFundamentals and order\nflow mechanics are conflicting;\ninstitutional participation\nis fragmented, leading to\npremium hunting on both\nsides of the market before\na decisive expansion."
    projected_pips := math.round(current_atr_pips * 0.5)
    direction_text := "REMAINING RANGE BOUND\n(NEUTRAL)"

// ==========================================
// DASHBOARD RENDERING (STRICT SQUARE DESIGN UI)
// ==========================================
// Strict black background configuration
color dark_bg = color.new(color.black, 5)
var table info_table = table.new(position.top_right, 1, 3, bgcolor=dark_bg, border_color=color.new(color.gray, 50), border_width=2)

if barstate.islast
    // Header block
    table.cell(info_table, 0, 0, "MARKET MOVER REPORT", text_color=color.white, text_size=size.small, bgcolor=color.new(color.blue, 20))
    
    // Core report block - Tight width (18) keeps it perfectly square
    table.cell(info_table, 0, 1, bias_report, text_color=color.white, text_size=size.small, width=18, bgcolor=dark_bg)
    
    // Directional and Pip movement block 
    table.cell(info_table, 0, 2, "EXPECTED MOVEMENT:\n" + str.tostring(projected_pips) + " PIPS\n" + direction_text, text_color=color.white, text_size=size.small, text_font_family=font.family_monospace, bgcolor=dark_bg)

// On-chart visual aids for sweeps
plotshape(sweep_low, title="Inst. Long Sweep", style=shape.triangleup, location=location.belowbar, color=color.green, size=size.small)
plotshape(sweep_high, title="Inst. Short Sweep", style=shape.triangledown, location=location.abovebar, color=color.red, size=size.small)
