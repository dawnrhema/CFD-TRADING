//@version=6
indicator("Most Pips (Time) - GMT Adjustable", overlay = true)

// ==========================================
// USER INPUTS & SETTINGS
// ==========================================
int lookbackDays = input.int(90, title="Lookback Period (Days)", minval=1, tooltip="Adjust the number of calendar/trading days to analyze for volatility averages.")
string tzInput   = input.string("GMT+0", title="Timezone", tooltip="Adjust this to your preferred GMT offset. Examples: 'GMT+0' (UK Winter), 'GMT+1' (UK Summer / BST), 'America/New_York', etc.")

// 1. Create a User-Defined Type (UDT) to manage slot history tracking
type SlotData
    string key
    float  totalDiff
    int    barCount

// Initialize a persistent array to store unique recurring time/day buckets
var SlotData[] registry = array.new<SlotData>()

// Identify the current candle's open-close absolute physical range
float candleBody = math.abs(open - close)

// 2. Dynamic Lookback Calculation based on active chart timeframe
// FIXED: Brought the ternary operator onto a single line to prevent the compiler error
int minutesPerBar = timeframe.isintraday ? timeframe.multiplier : 1440
int barsPerDay    = math.max(1, math.floor(1440 / minutesPerBar))
int totalBarsToScan = barsPerDay * lookbackDays

// Adjust specific edge cases for higher timeframes
if timeframe.isdaily
    totalBarsToScan := lookbackDays
else if timeframe.isweekly
    totalBarsToScan := math.max(1, math.floor(lookbackDays / 7))  // Scales days into weeks
else if timeframe.ismonthly
    totalBarsToScan := math.max(1, math.floor(lookbackDays / 30)) // Scales days into months

// Calculate the structural bar index distance from the real-time candle
int barsBack = last_bar_index - bar_index

// 3. Dynamically generate the grouping key
string slotKey = ""

if timeframe.isintraday
    // Extract hour and minute forced to the user-defined timezone
    int tzHour   = hour(time, tzInput)
    int tzMinute = minute(time, tzInput)
    
    string hhStr = tzHour < 10 ? "0" + str.tostring(tzHour) : str.tostring(tzHour)
    string mmStr = tzMinute < 10 ? "0" + str.tostring(tzMinute) : str.tostring(tzMinute)
    slotKey := hhStr + ":" + mmStr
else if timeframe.isdaily
    slotKey := "Day " + str.tostring(dayofmonth(time, tzInput))
else if timeframe.isweekly
    slotKey := "Week No. " + str.tostring(weekofyear(time, tzInput))
else
    slotKey := "Month No. " + str.tostring(month(time, tzInput))

// 4. Data Collection: Only scan within the target trading bar depth
if barsBack <= totalBarsToScan
    int foundIdx = -1
    
    if array.size(registry) > 0
        for i = 0 to array.size(registry) - 1
      
            if array.get(registry, i).key == slotKey
                foundIdx := i
                break
                
    if foundIdx != -1
        SlotData item = array.get(registry, foundIdx)
        item.totalDiff := item.totalDiff + candleBody
        item.barCount := item.barCount + 1
        array.set(registry, foundIdx, item)
    else
        array.push(registry, SlotData.new(slotKey, candleBody, 1))

// 5. Final Analytics Structuring, Sorting, and Counting
type FinalRank
    string name
    float  calculatedAvg
    int    count

if barstate.islast
    FinalRank[] rankList = array.new<FinalRank>()
    
    if array.size(registry) > 0
        for i = 0 to array.size(registry) - 1
           
            SlotData sd = array.get(registry, i)
            float finalAvg = sd.totalDiff / sd.barCount
            array.push(rankList, FinalRank.new(sd.key, finalAvg, sd.barCount))
            
    int n = array.size(rankList)
    if n > 1
        for i = 0 to n - 2
            for j = i + 1 to n - 1
 
                FinalRank r1 = array.get(rankList, i)
                FinalRank r2 = array.get(rankList, j)
                if r2.calculatedAvg > r1.calculatedAvg
                    array.set(rankList, i, r2)
                    array.set(rankList, j, r1)
                    
    // 6. Build the Visual Metrics Table Interface
    var table tb = table.new(position.top_right, 4, 8, bgcolor = color.black, border_color = color.gray, border_width = 1)
    
    string columnHeader = timeframe.isintraday ? "Candle Time Slot" : (timeframe.isdaily ? "Calendar Day" : "Period Slot")
    string averageHeader = str.tostring(lookbackDays) + "-Day Avg Size"
    
    // Custom dark gray color block for header background style
    color headerBg = color.rgb(40, 40, 40)
    
    // Header cells styling
    table.cell(tb, 0, 0, "Rank", text_color = color.white, bgcolor = headerBg)
    table.cell(tb, 1, 0, columnHeader, text_color = color.white, bgcolor = headerBg)
    table.cell(tb, 2, 0, averageHeader, text_color = color.white, bgcolor = headerBg)
    table.cell(tb, 3, 0, "Candles Scanned", text_color = color.white, bgcolor = headerBg)
    
    int maxDisplayRows = math.min(7, array.size(rankList))
    if maxDisplayRows > 0
        for i = 0 to maxDisplayRows - 1
            FinalRank fr = array.get(rankList, i)
            string printAvg = str.tostring(fr.calculatedAvg, "#.#####")
            
            // Data cells styling
            table.cell(tb, 0, i + 1, str.tostring(i + 1), text_color = color.white)
            table.cell(tb, 1, i + 1, fr.name, text_color = color.white)
            table.cell(tb, 2, i + 1, printAvg, text_color = color.white)
            table.cell(tb, 3, i + 1, str.tostring(fr.count), text_color = color.white)
