//@version=6
indicator("Orderflow Levels (CVD < 5% average CVD)", overlay = true, max_labels_count = 500, max_lines_count = 500, max_bars_back = 5000)

// --- Inputs ---
string ltf_input = input.timeframe("1", title = "Lower Timeframe for Delta", tooltip = "Must be lower than chart timeframe.")
color bullColor  = input.color(color.green, title = "Bullish Text Color")
color bearColor  = input.color(color.red, title = "Bearish Text Color")

// --- Your Requested Custom Inputs ---
int lookback     = input.int(500000, title = "CVD Average Lookback Window", minval = 10, tooltip = "Set to 500,000 to capture all available chart history.")
float pctCap     = input.float(4.0, title = "Volume Delta % Threshold", minval = 0.001, maxval = 500.0, step = 0.5, tooltip = "Triggers a line if current candle delta is under 4% of the historical average.")
color lineColor  = input.color(color.black, title = "Horizontal Line Color")
int lineWidth    = input.int(1, title = "Horizontal Line Width", minval = 1, maxval = 5)

// --- Volume & Delta Calculations ---
calc_fallback_delta() =>
    candleRange = high - low
    buyRatio  = candleRange == 0 ? 0.5 : (close - low) / candleRange
    sellRatio = candleRange == 0 ? 0.5 : (high - close) / candleRange
    (nz(volume) * buyRatio) - (nz(volume) * sellRatio)

float[] ltf_opens   = request.security_lower_tf(syminfo.tickerid, ltf_input, open)
float[] ltf_closes  = request.security_lower_tf(syminfo.tickerid, ltf_input, close)
float[] ltf_volumes = request.security_lower_tf(syminfo.tickerid, ltf_input, volume)

float deltaVol = 0.0
float totalVol = nz(volume)

if not na(ltf_closes) and array.size(ltf_closes) > 0
    for i = 0 to array.size(ltf_closes) - 1
        float l_open  = array.get(ltf_opens, i)
        float l_close = array.get(ltf_closes, i)
        float l_vol   = array.get(ltf_volumes, i)
        if l_close > l_open
            deltaVol += l_vol
        else if l_close < l_open
            deltaVol -= l_vol
else
    deltaVol := calc_fallback_delta()

float buyVol  = (totalVol + deltaVol) / 2
float sellVol = totalVol - buyVol
float cvd     = ta.cum(deltaVol)

// --- Smart Expanding Average Engine ---
var float rollingDeltaSum = 0.0
rollingDeltaSum := rollingDeltaSum + math.abs(deltaVol)

// If lookback is within standard TV limits (<=5000), use moving window truncation.
// If lookback is ultra-large (like 500k), treat it as an all-history expanding baseline.
if lookback <= 5000 and bar_index >= lookback
    rollingDeltaSum := rollingDeltaSum - math.abs(deltaVol[lookback])

int actualLookback = lookback <= 5000 ? int(math.min(bar_index + 1, lookback)) : bar_index + 1
float avgCandleDelta = rollingDeltaSum / actualLookback

// --- Threshold Condition Evaluation (< 4% of Average) ---
float currentMagnitude = math.abs(deltaVol)
float maxAllowedMagnitude = avgCandleDelta * (pctCap / 100.0)

// Triggers strictly when volume delta is essentially non-existent (dead zone tracking)
bool lineCondition = currentMagnitude < maxAllowedMagnitude

// --- Draw Fixed, Immovable Black Horizontal Lines ---
if lineCondition
    line.new(
         x1 = bar_index, y1 = close, 
         x2 = bar_index + 1, y2 = close, 
         xloc = xloc.bar_index, 
         extend = extend.right, // Infinite horizontal projection to the right
         color = lineColor, 
         width = lineWidth, 
         style = line.style_solid
     )

// --- Visuals (Labels below candles) ---
if totalVol > 0
    string txt = "Tot: " + str.tostring(totalVol, format.volume) + "\n" +
                 "Δ: "   + str.tostring(deltaVol, format.volume)
                 
    color textColor = deltaVol >= 0 ? bullColor : bearColor
    
    label.new(
         x = bar_index, 
         y = low, 
         text = txt, 
         style = label.style_label_up, 
         color = color.new(color.black, 100), 
         textcolor = textColor, 
         yloc = yloc.belowbar, 
         size = size.small
     )

// --- Dashboard Table Engine ---
var table mainTable = table.new(position.top_right, 2, 5, bgcolor=color.new(color.black, 30), border_color=color.gray, border_width=1)

if barstate.islast
    float avgVol = ta.sma(totalVol, 20)
    bool isHighVol = totalVol > avgVol
    bool isLowVol = totalVol < (avgVol * 0.7)
    
    string candleSignal = "Neutral / Balanced"
    color cellBgColor = color.gray
    
    if isHighVol
        if close > open and deltaVol < 0
            candleSignal := "Bullish Absorption (Sellers Trapped)"
            cellBgColor := color.blue
        else if close < open and deltaVol > 0
            candleSignal := "Bearish Absorption (Buyers Trapped)"
            cellBgColor := color.orange
        else if close > open and deltaVol > 0
            candleSignal := "Bullish Initiative (Aggressive Buying)"
            cellBgColor := color.green
        else if close < open and deltaVol < 0
            candleSignal := "Bearish Initiative (Aggressive Selling)"
            cellBgColor := color.red
    else if isLowVol
        candleSignal := "Exhaustion (Weak Interest)"
        cellBgColor := color.navy

    float cvdChange = cvd - cvd[14]
    string chartTrend = cvdChange > 0 ? "Aggressive Buyers Control" : "Aggressive Sellers Control"
    color trendColor = cvdChange > 0 ? color.green : color.red

    string conclusion = "Wait for Key Level"
    color concColor = color.white
    
    if candleSignal == "Bullish Initiative (Aggressive Buying)" and cvdChange > 0
        conclusion := "Strong Breakout / Continuation Up"
        concColor := color.green
    else if candleSignal == "Bearish Initiative (Aggressive Selling)" and cvdChange < 0
        conclusion := "Strong Breakdown / Continuation Down"
        concColor := color.red
    else if str.contains(candleSignal, "Absorption")
        conclusion := "Reversal Likely (Level is Holding)"
        concColor := color.yellow
    else if candleSignal == "Exhaustion (Weak Interest)"
        conclusion := "Fake-out Warning / Range Bound"
        concColor := color.orange

    table.cell(mainTable, 0, 0, "Orderflow Metric", text_color=color.white, text_size=size.small, bgcolor=color.black)
    table.cell(mainTable, 1, 0, "Analysis / State", text_color=color.white, text_size=size.small, bgcolor=color.black)
    table.cell(mainTable, 0, 1, "Last Candle State", text_color=color.white, text_size=size.small)
    table.cell(mainTable, 1, 1, candleSignal, text_color=color.white, text_size=size.small, bgcolor=cellBgColor)
    table.cell(mainTable, 0, 2, "General Trend (CVD)", text_color=color.white, text_size=size.small)
    table.cell(mainTable, 1, 2, chartTrend, text_color=trendColor, text_size=size.small)
    table.cell(mainTable, 0, 3, "Total Cumulative Delta", text_color=color.white, text_size=size.small)
    table.cell(mainTable, 1, 3, str.tostring(cvd, format.volume), text_color=color.white, text_size=size.small)
    table.cell(mainTable, 0, 4, "Market Conclusion", text_color=color.black, text_size=size.small, bgcolor=color.yellow)
    table.cell(mainTable, 1, 4, conclusion, text_color=color.white, text_size=size.small, bgcolor=color.new(concColor, 20))
