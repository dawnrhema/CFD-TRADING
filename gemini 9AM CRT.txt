//@version=6
indicator("9AM London Anchor Candle Range Model (Permanent)", overlay=true, max_boxes_count=500, max_lines_count=500)

// --- Inputs ---
string t_group   = "Time Settings (London Time)"
string i_tz      = input.string("Europe/London", "Exchange/Session Timezone", group=t_group)
int i_start_hour = input.int(9, "Anchor Candle Hour (9AM)", minval=0, maxval=23, group=t_group)
int i_sweep_hour = input.int(10, "Sweep/Execution Hour Starts (10AM)", minval=0, maxval=23, group=t_group)

string c_group   = "Style & Colors"
color c_range    = input.color(color.new(color.purple, 93), "9AM Range Fill", group=c_group)
color c_high     = input.color(color.purple, "9AM High Line", group=c_group)
color c_low      = input.color(color.orange, "9AM Low Line", group=c_group)
color c_sweep_bg = input.color(color.new(color.purple, 97), "Execution Window Background", group=c_group)

// --- Time Calculations ---
int current_hour = hour(time, i_tz)

// Flags to identify the specific hours
bool is_anchor_hour  = current_hour == i_start_hour
bool is_sweep_window = current_hour >= i_sweep_hour

// --- Persistent State Variables ---
var float anchor_high = na
var float anchor_low  = na
var int anchor_start_time = na

// Arrays to store history so drawings are never deleted
var box[] historical_boxes = array.new_box()
var line[] historical_high_lines = array.new_line()
var line[] historical_low_lines = array.new_line()

var bool sweep_high_triggered = false
var bool sweep_low_triggered  = false

// --- Logic ---
// Detect the exact start of a new 9:00 AM anchor hour
if is_anchor_hour and not (is_anchor_hour[1])
    anchor_high := high
    anchor_low := low
    anchor_start_time := time
    sweep_high_triggered := false
    sweep_low_triggered := false
else if is_anchor_hour
    // Dynamically lock the absolute High/Low during the 9:00 AM hour
    anchor_high := math.max(anchor_high, high)
    anchor_low := math.min(anchor_low, low)

// When the 9:00 AM candle closes and the 10:00 AM window begins
if is_sweep_window and not (is_sweep_window[1])
    // Create new permanent visual elements for today
    box new_box = box.new(left=anchor_start_time, top=anchor_high, right=time, bottom=anchor_low, 
                          xloc=xloc.bar_time, bgcolor=c_range, border_color=color.new(color.purple, 60), border_style=line.style_dashed)
    
    line new_high = line.new(x1=anchor_start_time, y1=anchor_high, x2=time, y2=anchor_high, 
                             xloc=xloc.bar_time, color=c_high, width=1)
                          
    line new_low  = line.new(x1=anchor_start_time, y1=anchor_low, x2=time, y2=anchor_low, 
                             xloc=xloc.bar_time, color=c_low, width=1)
    
    // Push them to arrays to preserve them in chart history
    array.push(historical_boxes, new_box)
    array.push(historical_high_lines, new_high)
    array.push(historical_low_lines, new_low)

// Update the most recent active drawings in real-time as the current day progresses
if is_sweep_window and array.size(historical_boxes) > 0
    box active_box = array.get(historical_boxes, array.size(historical_boxes) - 1)
    line active_high = array.get(historical_high_lines, array.size(historical_high_lines) - 1)
    line active_low = array.get(historical_low_lines, array.size(historical_low_lines) - 1)
    
    box.set_right(active_box, time)
    line.set_x2(active_high, time)
    line.set_x2(active_low, time)

// --- Detection of the Liquidity Raid (Post-9AM Range) ---
if is_sweep_window
    // Bearish Sweep: Price breaks above the 9AM High
    if high > anchor_high and not sweep_high_triggered
        sweep_high_triggered := true
        label.new(x=time, y=high, text="9AM High Swept\n(Look for MSS)", xloc=xloc.bar_time, color=color.red, textcolor=color.white, style=label.style_label_down)
        
    // Bullish Sweep: Price breaks below the 9AM Low
    if low < anchor_low and not sweep_low_triggered
        sweep_low_triggered := true
        label.new(x=time, y=low, text="9AM Low Swept\n(Look for MSS)", xloc=xloc.bar_time, color=color.green, textcolor=color.white, style=label.style_label_up)

// --- Visual Background Styling for Execution Window ---
bgcolor(is_sweep_window ? c_sweep_bg : na)
