//@version=6
indicator("8AM London 1H Anchor Candle Range Model", overlay=true, max_boxes_count=500, max_lines_count=500)

// --- Inputs ---
string t_group   = "Time Settings (London Time)"
string i_tz      = input.string("Europe/London", "Exchange/Session Timezone", group=t_group)
int i_start_hour = input.int(8, "Anchor Candle Hour (8AM)", minval=0, maxval=23, group=t_group)
int i_sweep_hour = input.int(9, "Sweep/Execution Hour Starts (9AM)", minval=0, maxval=23, group=t_group)
int i_end_hour   = input.int(12, "Box Extension End Hour (12PM)", minval=0, maxval=23, group=t_group)

string c_group   = "Style & Colors"
color c_range    = input.color(color.new(color.purple, 80), "8AM Range Fill", group=c_group)
color c_high     = input.color(color.purple, "8AM High Line", group=c_group)
color c_low      = input.color(color.orange, "8AM Low Line", group=c_group)
color c_sweep_bg = input.color(color.new(color.purple, 88), "Execution Window Background", group=c_group)

// --- 1-Hour Timeframe Calculations ---
f_1h_anchor(i_end) =>
    s_hour       = hour(time, i_tz)
    is_anchor    = s_hour == i_start_hour
    is_sweep     = s_hour >= i_sweep_hour
    is_active    = s_hour >= i_sweep_hour and s_hour < i_end
    
    var float a_high = na
    var float a_low  = na
    var int a_time   = na
    
    if is_anchor and not is_anchor[1]
        a_high := high
        a_low  := low
        a_time := time
    else if is_anchor
        a_high := math.max(a_high, high)
        a_low  := math.min(a_low, low)
        
    [a_high, a_low, a_time, is_sweep, is_active]

// Fetch data from the 1-hour context
[htf_high, htf_low, htf_time, htf_sweep, htf_active] = request.security(syminfo.tickerid, "60", f_1h_anchor(i_end_hour), gaps=barmerge.gaps_off)

// --- Persistent State Variables ---
var box[] historical_boxes = array.new_box()
var line[] historical_high_lines = array.new_line()
var line[] historical_low_lines = array.new_line()

var bool sweep_high_triggered = false
var bool sweep_low_triggered  = false

bool sweep_window_started = htf_sweep and not htf_sweep[1]
bool new_anchor_started   = not htf_sweep and htf_sweep[1]

if new_anchor_started
    sweep_high_triggered := false
    sweep_low_triggered  := false

// --- Logic ---
// When the 8:00 AM 1H candle closes and the 9:00 AM window begins
if sweep_window_started
    box new_box = box.new(left=htf_time, top=htf_high, right=time, bottom=htf_low, 
                          xloc=xloc.bar_time, bgcolor=c_range, border_color=color.new(color.purple, 40), border_style=line.style_dashed, border_width=2)
    
    line new_high = line.new(x1=htf_time, y1=htf_high, x2=time, y2=htf_high, 
                             xloc=xloc.bar_time, color=c_high, width=3)
                          
    line new_low  = line.new(x1=htf_time, y1=htf_low, x2=time, y2=htf_low, 
                             xloc=xloc.bar_time, color=c_low, width=3)
    
    array.push(historical_boxes, new_box)
    array.push(historical_high_lines, new_high)
    array.push(historical_low_lines, new_low)

// ONLY update and extend the drawings if we are within the active window (before 12:00 PM London)
if htf_active and array.size(historical_boxes) > 0
    box active_box = array.get(historical_boxes, array.size(historical_boxes) - 1)
    line active_high = array.get(historical_high_lines, array.size(historical_high_lines) - 1)
    line active_low = array.get(historical_low_lines, array.size(historical_low_lines) - 1)
    
    box.set_right(active_box, time)
    line.set_x2(active_high, time)
    line.set_x2(active_low, time)

// --- Detection of the Liquidity Raid ---
if htf_sweep
    if high > htf_high and not sweep_high_triggered
        sweep_high_triggered := true
        label.new(x=time, y=high, text="8AM 1H High Swept\n(Look for MSS)", xloc=xloc.bar_time, color=color.red, textcolor=color.white, style=label.style_label_down)
         
    if low < htf_low and not sweep_low_triggered
        sweep_low_triggered := true
        label.new(x=time, y=low, text="8AM 1H Low Swept\n(Look for MSS)", xloc=xloc.bar_time, color=color.green, textcolor=color.white, style=label.style_label_up)

// --- Visual Background Styling ---
bgcolor(htf_sweep ? c_sweep_bg : na)
